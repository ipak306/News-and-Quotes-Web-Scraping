def scrape_quote(data):
    quote = []
    desc = data.findAll('span',{'class':'text'}) # tag html  class = text   quote 
    #  quote  list quote
    for a in range(0,len(desc)):
        quote.append(desc[a].text.replace('\'','').replace('','').replace('',''))
        
    return quote

def lemma_tokenizer_w_pos_tag(text):
    
    #  POS tag   Lemmatizer 
    def convert_tags(tag):
        if tag == 'vbd' or tag == 'vbg' or tag == 'vbz': # tag = vbd,vbg,vbz  return  v(verb)
            return 'v'
        else:  # return  n(noun)
            return 'n'

    # standard_tokenizer
    standard_tokenizer = CountVectorizer().build_tokenizer()
    #print(standard_tokenizer)
    tokens = standard_tokenizer(text)
    #print(tokens)
    
    # pos tag 
    tokens_with_pos_tag = nltk.pos_tag(tokens)
    #print(tokens_with_pos_tag)
    
    #  NLTK  lemmatisation()
    lemmatizer = nltk.WordNetLemmatizer()
    lemma_tokens = []
    
    for token in tokens_with_pos_tag: # token
    #print(token)
    #print(token[1].lower())
        new_tag = convert_tags(token[1].lower())  # convert_tags  POS tag 
    #print(new_tag)
    #dd = lemmatizer.lemmatize(token[0]) 
    #print(dd)
        lemma_tokens.append(lemmatizer.lemmatize(token[0], new_tag)) # lemma_tokens
    
    #print(lemma_tokens)
    return lemma_tokens

def create_model(model, parameters,feature):
    total2 = [] #  list  accuracy precision  recall  tags
    
    for i in label_top15: #  tags 15 
        X_train , X_test , y_train , y_test = train_test_split(feature, label_top15[i], test_size = 0.1, random_state = 0)
        #  2  trian = 90 test = 10 
        
        total = []
        
        # best parameter  GridsearchCV 
        skf = StratifiedKFold(n_splits=5, shuffle = True, random_state = 1001)
        #  model   parameters  model 
        tune = GridSearchCV(model, parameters, verbose = 3, cv = skf.split(X_train,y_train), scoring='accuracy', n_jobs = -1)
        tune.fit(X_train, y_train)

        bestparams = tune.best_params_
        model.set_params(**bestparams) # set  best parameter  GridSearchCV  model

        model.fit(X_train, y_train) # model X_train, y_train
        predicted = model.predict(X_test) #  model  X_test

        accuracy = accuracy_score(y_test, predicted) #  y_test  predicted  accuracy

        precision = precision_score(y_test,predicted, average='binary') #  precision #binary  precision 

        recall = recall_score(y_test,predicted, average='binary') #  recall #binary  precision 
        
        # accuracy precision recall  tags  list total
        total.append('%.2f'%accuracy) 
        total.append('%.2f'%precision)
        total.append('%.2f'%recall)
        total2.append(total) # accuracy precision recall  tags  list total2
    
    #  list total2  dataframe 
    df = pd.DataFrame(total2, index=[i for i in label_top15],columns = ['accuracy','precision','recall']) 
    
    return  df
